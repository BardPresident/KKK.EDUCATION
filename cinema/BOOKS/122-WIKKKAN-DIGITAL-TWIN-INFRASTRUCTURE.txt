## PART FOUR: WIKKKAN AI — HOW IT WORKS

### The Architecture

```

User types a question at kkk.education/wikkkan.html

↓

HTTPS POST to wikkkan.kkk.education/wikkkan

↓

Nginx (port 443) → Gunicorn (port 5000)

↓

Flask API — wikkkan.py

↓

Sentence Transformers converts question to embedding vector

↓

ChromaDB searches 24,514 book chunks for relevant passages

↓

Top 3 most relevant chunks sent as context to language model

↓

Ollama + phi3:mini generates answer in WIKKKAN's voice

↓

Answer + source book names returned to browser

```

### The Knowledge Base

All 121 books were processed into a searchable vector database:

- Each book split into 1,000-character chunks

- Each chunk converted to a 384-dimensional embedding vector

- Stored in ChromaDB at `/home/wikkkan/chromadb/`

- Total: 24,514 chunks across 121 books

- Embedding model: all-MiniLM-L6-v2 (sentence-transformers)

When a question arrives, the question is converted to the same vector format and the database finds the most similar book passages — semantic search, not keyword search. It finds meaning, not just words.

### The Language Model

- **Current model:** phi3:mini (Microsoft)

- **Why phi3:** Fastest on CPU, good quality for a small model

- **Runs via:** Ollama (local model server)

- **No GPU required:** Runs entirely on CPU

- **No internet required:** Completely self-contained

### The Voice

The prompt instructs WIKKKAN to speak as the embodiment of Wendell Charles NeSmith — drawing only from the retrieved book passages, never inventing quotes or book titles, keeping answers complete and concise, never adding safety disclaimers.

index_book.py


import chromadb
import requests
import json
import hashlib
from sentence_transformers import SentenceTransformer

# ── CONFIG ────────────────────────────────────────────────────────
MANIFEST_URL = 'https://kkk.education/manifest.json'
BOOKS_FOLDER = 'BOOKS'
CHROMA_DIR   = '/home/wikkkan/chromadb'
CHUNK_SIZE   = 1000

# ── SETUP ─────────────────────────────────────────────────────────
print("Loading embedding model...")
model  = SentenceTransformer('all-MiniLM-L6-v2')
client = chromadb.PersistentClient(path=CHROMA_DIR)
col    = client.get_or_create_collection("wikkkan_books")

# ── FETCH MANIFEST ────────────────────────────────────────────────
print("Fetching manifest from kkk.education...")
res      = requests.get(MANIFEST_URL)
manifest = res.json()

# ── FIND BOOKS FOLDER ─────────────────────────────────────────────
def find_folder(items, name):
    for item in items:
        if item.get('type') == 'folder' and item.get('name') == name:
            return item
        if item.get('type') == 'folder':
            result = find_folder(item.get('children', []), name)
            if result:
                return result
    return None

# ── COLLECT TXT FILES ─────────────────────────────────────────────
def collect_txt(items):
    files = []
    for item in items:
        if item.get('type') == 'folder':
            files += collect_txt(item.get('children', []))
        elif item.get('name', '').lower().endswith('.txt'):
            files.append({'name': item['name'], 'url': item['path']})
    return files

books_folder = find_folder(manifest['items'], BOOKS_FOLDER)
if not books_folder:
    print("ERROR: BOOKS folder not found in manifest!")
    exit(1)

txt_files = collect_txt(books_folder.get('children', []))
print(f"Found {len(txt_files)} books in manifest.")

# ── CONTENT HASH HELPER ────────────────────────────────────────────
def content_hash(text):
    return hashlib.md5(text.encode('utf-8')).hexdigest()

def get_stored_hash(book_name):
    """Get the hash stored in the first chunk's metadata, if any."""
    first_id = book_name + '_0'
    existing = col.get(ids=[first_id], include=['metadatas'])
    if existing['ids']:
        return existing['metadatas'][0].get('content_hash', None)
    return None

def delete_book_chunks(book_name):
    """Delete all chunks for a given book."""
    results = col.get(where={'source': book_name}, include=['metadatas'])
    if results['ids']:
        col.delete(ids=results['ids'])
        return len(results['ids'])
    return 0

# ── INDEX EACH BOOK ───────────────────────────────────────────────
indexed  = 0
updated  = 0
skipped  = 0
errors   = 0

for f in txt_files:
    name = f['name']
    url  = f['url']

    if url.startswith('/'):
        url = 'https://kkk.education' + url

    print(f"\n📖 {name}")

    try:
        r = requests.get(url, timeout=120)
        r.raise_for_status()
        text = r.text
    except Exception as e:
        print(f"  ❌ Failed to fetch: {e}")
        errors += 1
        continue

    # Compute hash of current content
    current_hash = content_hash(text)

    # Check if already indexed with same content
    stored_hash = get_stored_hash(name)

    if stored_hash is not None:
        if stored_hash == current_hash:
            print(f"  ⏭  Unchanged — skipping")
            skipped += 1
            continue
        else:
            # Content changed — delete old chunks and reindex
            deleted = delete_book_chunks(name)
            print(f"  🔄 Content changed — deleted {deleted} old chunks, reindexing...")
            updated += 1
    else:
        print(f"  ✨ New book — indexing...")
        indexed += 1

    # Index chunks
    chunks = [text[i:i+CHUNK_SIZE] for i in range(0, len(text), CHUNK_SIZE)]
    print(f"  {len(chunks)} chunks...")

    for j, chunk in enumerate(chunks):
        chunk_id  = name + '_' + str(j)
        embedding = model.encode(chunk).tolist()

        # Store content_hash in metadata of every chunk (for easy lookup via chunk 0)
        col.add(
            ids=[chunk_id],
            embeddings=[embedding],
            documents=[chunk],
            metadatas=[{
                "source":       name,
                "chunk":        j,
                "content_hash": current_hash
            }]
        )

    print(f"  ✅ Done")

print(f"\n{'='*50}")
print(f"✨ New books indexed: {indexed}")
print(f"🔄 Updated books reindexed: {updated}")
print(f"⏭  Unchanged books skipped: {skipped}")
print(f"❌ Errors: {errors}")
print(f"📚 Total chunks in ChromaDB: {col.count()}")

# ── Title keyword extraction ──────────────────────────────────────────────────
# Pulls capitalised multi-word phrases or quoted strings from the question
# so we can do a direct metadata filter rather than relying on semantic match.

def extract_title_keywords(question):
    """Return a list of candidate title fragments from the question."""
    candidates = []

    # Quoted strings
    candidates += re.findall(r'"([^"]+)"', question)
    candidates += re.findall(r"'([^']+)'", question)

    # Long capitalised runs (3+ consecutive Title-Case or ALL-CAPS words)
    caps_run = re.findall(r'(?:[A-Z][a-zA-Z]+[\s\-]+){2,}[A-Z][a-zA-Z]+', question)
    candidates += caps_run

    # Any word that looks like a book number prefix  e.g. "124" or "book 124"
    nums = re.findall(r'\b(\d{2,3})\b', question)
    candidates += nums

    return [c.strip() for c in candidates if len(c.strip()) > 3]


def search_by_title(keywords, n=6):
    """Fetch chunks whose source filename contains any of the keywords."""
    if not keywords:
        return [], []

    # ChromaDB `where` only supports exact/contains on string fields via $contains
    # We pull all metadata and filter in Python (fast enough for typical corpus sizes)
    all_meta = col.get(include=['metadatas', 'documents'])
    chunks, sources = [], []

    for doc, meta in zip(all_meta['documents'], all_meta['metadatas']):
        src = meta.get('source', '').lower()
        for kw in keywords:
            if kw.lower().replace(' ', '-') in src or kw.lower().replace(' ', '_') in src or kw.lower() in src:
                if doc not in chunks:
                    chunks.append(doc)
                    sources.append(meta['source'])
                break
        if len(chunks) >= n:
            break

    return chunks, sources


# ── Semantic search ───────────────────────────────────────────────────────────

def search_books(question, n=8):
    embedding = model.encode(question).tolist()
    results   = col.query(
        query_embeddings=[embedding],
        n_results=n,
        include=['documents', 'metadatas', 'distances']
    )
    chunks, sources = [], []
    for doc, meta, dist in zip(
        results['documents'][0],
        results['metadatas'][0],
        results['distances'][0]
    ):
        if dist < 1.1:
            chunks.append(doc)
            sources.append(meta['source'])
    return chunks, sources


# ── Web tools ─────────────────────────────────────────────────────────────────

def search_web(query):
    try:
        with DDGS() as ddgs:
            results = list(ddgs.text(query, max_results=4))
        if not results:
            return "No web results found.", []
        urls = [r.get('href', '') for r in results if r.get('href')]
        text = '\n\n'.join([
            f"SOURCE: {r.get('href','unknown')}\nTITLE: {r['title']}\nSUMMARY: {r['body']}"
            for r in results
        ])
        return text, urls
    except Exception as e:
        return f"Web search failed: {e}", []


def fetch_url(url):
    try:
        r = requests.get(url, timeout=15, headers={'User-Agent': 'Mozilla/5.0'})
        soup = BeautifulSoup(r.text, 'html.parser')
        for tag in soup(['script','style','nav','footer','header','aside']):
            tag.decompose()
        return soup.get_text(separator='\n', strip=True)[:5000]
    except Exception as e:
        return f"Could not fetch URL: {e}"

@app.route('/wikkkan', methods=['POST'])
def wikkkan():
    data     = request.json
    question = data.get('question', '').strip()
    use_web  = data.get('web', False)
    url      = data.get('url', '')

    if not question:
        return jsonify({'error': 'No question provided'}), 400

    context_parts = []
    web_urls      = []
    sources       = []
    chunks        = []

    # ── 1. Title keyword search (catches "do you have X", "tell me about book X") ──
    title_keywords = extract_title_keywords(question)
    if title_keywords:
        title_chunks, title_sources = search_by_title(title_keywords)
        for c, s in zip(title_chunks, title_sources):
            if c not in chunks:
                chunks.append(c)
                sources.append(s)

    # ── 2. Semantic search ────────────────────────────────────────────────────
    sem_chunks, sem_sources = search_books(question)
    for c, s in zip(sem_chunks, sem_sources):
        if c not in chunks:
            chunks.append(c)
            sources.append(s)

    # ── 3. Keyword boosters for known proper nouns ────────────────────────────
    extra_queries = []
    q_lower = question.lower()
    if 'wendy' in q_lower:
        extra_queries += ['Wendy Ebur NeSmith Rainbow Bridge', 'Wendy memorial ivory']
    if 'rainbow bridge' in q_lower:
        extra_queries += ['Rainbow Bridge Wendy Ivory']
    if 'open source university' in q_lower:
        extra_queries += ['Open Source University Wendy']
    for eq in extra_queries:
        ec, es = search_books(eq, n=3)
        for c, s in zip(ec, es):
            if c not in chunks:
                chunks.append(c)
                sources.append(s)

    has_context = len(chunks) > 0

    if has_context:
        source_list = ', '.join(set(sources))
        context_parts.append(
            f"FROM WIKKKAN'S BOOKS ({source_list}):\n" +
            '\n---\n'.join(chunks)
        )

    if url:
        context_parts.append("FROM URL:\n" + fetch_url(url))

    if use_web:
        web_text, web_urls = search_web(question)
        context_parts.append("FROM WEB:\n" + web_text)

    context = '\n\n'.join(context_parts)
    system_msg, user_msg = build_messages(question, context, use_web, has_context)

    def generate():
        yield f"data: {json.dumps({'sources': list(set(sources)), 'web_urls': web_urls})}\n\n"

        stream = ollama.chat(
            model='qwen2.5:7b',
            options={
                'temperature': 0.3,
                'num_predict': 600,
                'repeat_penalty': 1.15,
                'top_p': 0.9,
            },
            messages=[
                {'role': 'system', 'content': system_msg},
                {'role': 'user',   'content': user_msg},
            ],
            stream=True
        )

        for chunk in stream:
            token = chunk['message']['content']
            if token:
                yield f"data: {json.dumps({'token': token})}\n\n"

        yield f"data: {json.dumps({'done': True})}\n\n"

    return Response(
        stream_with_context(generate()),
        mimetype='text/event-stream',
        headers={'Cache-Control': 'no-cache', 'X-Accel-Buffering': 'no'}
    )
